```swift
import Foundation
@testable import StrataClient

actor MockAPIClient: APIClientProtocol {
    var uploadAudioResult: Result<String, Error> = .success("job-123")
    var uploadURLResult: Result<String, Error> = .success("job-456")
    var pollResult: Result<JobResult, Error> = .success(JobResult(zipData: nil, status: "completed"))

    var uploadAudioCallCount = 0
    var uploadURLCallCount = 0
    var pollCallCount = 0

    func uploadAudio(fileData: Data, fileName: String, mimeType: String, token: String) async throws -> String {
        uploadAudioCallCount += 1
        return try uploadAudioResult.get()
    }

    func uploadURL(urlString: String, token: String) async throws -> String {
        uploadURLCallCount += 1
        return try uploadURLResult.get()
    }

    func pollJobStatus(jobId: String, token: String, intervalSeconds: Double, maxAttempts: Int) async throws -> JobResult {
        pollCallCount += 1
        return try pollResult.get()
    }
}
```

El pattern para tests de @MainActor @Observable:
- `await MainActor.run { ... }` para setup y assertions en contexto MainActor
- `Task.sleep` o `await fulfillment` para esperar que el Task interno del ViewModel termine

```swift
import Testing
import Foundation
@testable import StrataClient

@MainActor
struct ImportViewModelTests {

    // MARK: - Helpers

    func makeViewModel(
        mockClient: MockAPIClient = MockAPIClient(),
        existingHashes: [String] = []
    ) throws -> (ImportViewModel, MockAPIClient, LibraryStore) {
        let cacheManager = try CacheManager()
        let libraryStore = LibraryStore(cacheManager: cacheManager)
        // Pre-poblar LibraryStore con canciones existentes para cache hit tests
        // LibraryStore.songs es private(set) — necesitamos addSong para pre-popularlo
        // O usar una subclase mock. Más simple: usar un archivo real si existingHashes es vacío,
        // o crear SongEntry y llamar await libraryStore.addSong si hay hashes.
        // Para tests de cache hit usar una MockLibraryStore o inyectar songs directamente.

        let authViewModel = AuthViewModel()
        // Nota: AuthViewModel.token es necesario — en tests el token puede ser nil si no está logado.
        // ImportViewModel hace guard let token — en tests de cache hit no llegamos al upload,
        // así que token nil es aceptable.

        let viewModel = ImportViewModel(
            apiClient: mockClient,
            cacheManager: cacheManager,
            libraryStore: libraryStore,
            authViewModel: authViewModel
        )
        return (viewModel, mockClient, libraryStore)
    }

    // MARK: - Cache Hit Tests

    @Test func cacheHitFile() async throws {
        let (viewModel, mockClient, libraryStore) = try makeViewModel()
        // Crear un archivo temporal de prueba
        let tempFile = FileManager.default.temporaryDirectory
            .appendingPathComponent("test_cache_hit.mp3")
        let fakeAudioData = Data(repeating: 0xAA, count: 1024)
        try fakeAudioData.write(to: tempFile)
        defer { try? FileManager.default.removeItem(at: tempFile) }

        // Calcular hash real del archivo
        let cacheManager = try CacheManager()
        let hash = try await cacheManager.sha256(of: tempFile)

        // Pre-poblar la biblioteca con ese hash
        let existingEntry = SongEntry(
            id: UUID(), title: "Existing", artist: nil, duration: 30,
            sourceURL: nil, fileName: "test.mp3", sourceHash: hash, addedAt: Date()
        )
        await libraryStore.addSong(existingEntry)

        viewModel.startFileImport(from: tempFile)

        // Esperar que el Task interno complete
        try await Task.sleep(nanoseconds: 500_000_000) // 0.5s

        #expect(viewModel.phase == .ready(cached: true))
        let callCount = await mockClient.uploadAudioCallCount
        #expect(callCount == 0)
    }

    @Test func cacheHitURL() async throws {
        let (viewModel, mockClient, libraryStore) = try makeViewModel()
        let videoID = "dQw4w9WgXcQ"
        let youtubeURL = "https://www.youtube.com/watch?v=\(videoID)"

        let existingEntry = SongEntry(
            id: UUID(), title: "Cached YT Song", artist: nil, duration: 210,
            sourceURL: youtubeURL, fileName: nil, sourceHash: videoID, addedAt: Date()
        )
        await libraryStore.addSong(existingEntry)

        viewModel.startURLImport(urlString: youtubeURL)
        try await Task.sleep(nanoseconds: 200_000_000)

        #expect(viewModel.phase == .ready(cached: true))
        let callCount = await mockClient.uploadURLCallCount
        #expect(callCount == 0)
    }

    @Test func invalidYouTubeURL() async throws {
        let (viewModel, mockClient, _) = try makeViewModel()

        viewModel.startURLImport(urlString: "https://example.com/not-youtube")
        try await Task.sleep(nanoseconds: 200_000_000)

        if case .error(let msg) = viewModel.phase {
            #expect(msg.contains("YouTube"))
        } else {
            Issue.record("Expected .error phase, got \(viewModel.phase)")
        }
        let callCount = await mockClient.uploadURLCallCount
        #expect(callCount == 0)
    }

    @Test func uploadNetworkError() async throws {
        let mockClient = MockAPIClient()
        await mockClient.setUploadAudioResult(.failure(APIError.httpError(500)))
        let (viewModel, _, _) = try makeViewModel(mockClient: mockClient)

        let tempFile = FileManager.default.temporaryDirectory
            .appendingPathComponent("test_upload_error.mp3")
        try Data(repeating: 0xBB, count: 512).write(to: tempFile)
        defer { try? FileManager.default.removeItem(at: tempFile) }

        viewModel.startFileImport(from: tempFile)
        try await Task.sleep(nanoseconds: 500_000_000)

        if case .error = viewModel.phase { } else {
            Issue.record("Expected .error phase, got \(viewModel.phase)")
        }
    }

    @Test func pollError() async throws {
        let mockClient = MockAPIClient()
        await mockClient.setPollResult(.failure(APIError.processingFailed("stem separation failed")))
        let (viewModel, _, _) = try makeViewModel(mockClient: mockClient)

        let tempFile = FileManager.default.temporaryDirectory
            .appendingPathComponent("test_poll_error.mp3")
        try Data(repeating: 0xCC, count: 512).write(to: tempFile)
        defer { try? FileManager.default.removeItem(at: tempFile) }

        viewModel.startFileImport(from: tempFile)
        try await Task.sleep(nanoseconds: 500_000_000)

        if case .error(let msg) = viewModel.phase {
            #expect(msg.contains("stem separation"))
        } else {
            Issue.record("Expected .error phase, got \(viewModel.phase)")
        }
    }

    @Test func cancelReturnsToIdle() async throws {
        let (viewModel, _, _) = try makeViewModel()

        let tempFile = FileManager.default.temporaryDirectory
            .appendingPathComponent("test_cancel.mp3")
        try Data(repeating: 0xDD, count: 512).write(to: tempFile)
        defer { try? FileManager.default.removeItem(at: tempFile) }

        viewModel.startFileImport(from: tempFile)
        viewModel.cancel()
        try await Task.sleep(nanoseconds: 200_000_000)

        #expect(viewModel.phase == .idle)
    }
}
```
